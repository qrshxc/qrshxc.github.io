一、单总线协议的原理和应用

单总线协议指的是只采用一根信道来进行数据传输，通信指的是双方（MCU 与传感器）通过一根信道进行数据交互，所以按照数据的传输方向，只能采用半双工通信方式，比较典型的传感器就是温度传感器（DS18B20、DHT11……）

/**
 ******************************************************************************
 * @file    DHT11.c 
 * @author  qrshxc@163.com
 * @version V1.0
 * @date    2025.08.01
 * @brief   利用DHT11测试温湿度 PE0
 ******************************************************************************
 **/

#include "stm32f4xx.h"
#include <stdio.h>
#include <stdbool.h>

/**
 * @name      delay_us
 * @brief     微秒级延时函数
 * @param     nus: 延时的微秒数
 * @return    无
 * @version   1.0
 * @note      使用SysTick定时器实现精确微秒延时
 */
void delay_us(u32 nus)
{
	SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
	SysTick->LOAD = nus * 21 -1; // 计数次数，递减计数，0结束
	SysTick->VAL = 0;     // 清除当前数值寄存器
	SysTick->CTRL = 1;    // 打开系统嘀嗒定时器，使用参考时钟
	while ((SysTick->CTRL & 0x00010000)==0); // 等待计时完成
	SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
}

/**
 * @name      delay_ms
 * @brief     毫秒级延时函数
 * @param     nus: 延时的毫秒数
 * @return    无
 * @version   1.0
 * @note      使用SysTick定时器实现精确毫秒延时
 */
void delay_ms(u32 nus)
{
	SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
	SysTick->LOAD = nus * 21 * 1000 -1; // 计数次数，递减计数，0结束
	SysTick->VAL = 0;     // 清除当前数值寄存器
	SysTick->CTRL = 1;    // 打开系统嘀嗒定时器，使用参考时钟
	while ((SysTick->CTRL & 0x00010000)==0); // 等待计时完成
	SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
}

/**
 * @name      USART1_Config
 * @brief     配置USART1串口
 * @param     baud: 波特率
 * @return    无
 * @version   1.0
 * @note      配置USART1为异步模式，8位数据位，1位停止位，无校验
 */
void USART1_Config(uint32_t baud)
{
	USART_InitTypeDef USART_InitStructure;
	GPIO_InitTypeDef GPIO_InitStructure;

	/* 打开GPIOA时钟 */	
	RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOA, ENABLE);

	/* 打开USART1时钟 */
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_USART1, ENABLE);	
	
	/* 设置PA9和PA10的复用功能为USART1 */
	GPIO_PinAFConfig(GPIOA, GPIO_PinSource9, GPIO_AF_USART1);  // PA9 - USART1_TX
	GPIO_PinAFConfig(GPIOA, GPIO_PinSource10, GPIO_AF_USART1); // PA10 - USART1_RX
	
	/* 配置GPIO的复用功能+初始化 */
	GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;        // 复用功能模式
	GPIO_InitStructure.GPIO_Speed = GPIO_Speed_100MHz;  // 高速模式
	GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;      // 推挽输出
	GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;        // 上拉模式
	GPIO_InitStructure.GPIO_Pin = GPIO_Pin_9 | GPIO_Pin_10; // PA9(TX), PA10(RX)
	GPIO_Init(GPIOA, &GPIO_InitStructure);

	/* 配置USART1参数+初始化 */
	USART_InitStructure.USART_BaudRate = baud;          // 波特率
	USART_InitStructure.USART_WordLength = USART_WordLength_8b; // 8位数据位
	USART_InitStructure.USART_StopBits = USART_StopBits_1;     // 1位停止位
	USART_InitStructure.USART_Parity = USART_Parity_No;        // 无校验位
	USART_InitStructure.USART_HardwareFlowControl = USART_HardwareFlowControl_None; // 无硬件流控
	USART_InitStructure.USART_Mode = USART_Mode_Rx | USART_Mode_Tx; // 收发模式
	USART_Init(USART1, &USART_InitStructure);

	/* 打开USART1 */
	USART_Cmd(USART1, ENABLE);
}

/**
 * @name      USART1_SendString
 * @brief     通过USART1发送字符串
 * @param     str: 要发送的字符串
 * @return    无
 * @version   1.0
 * @note      循环发送字符串中的每个字符，等待发送完成
 */
void USART1_SendString(const char* str)
{
	while(*str)
	{
		USART_SendData(USART1, *str++);  // 发送字符并指针后移
		while(!USART_GetFlagStatus(USART1, USART_FLAG_TXE)); // 等待发送完成
	}
}

/**
 * @name      DHT11_Output
 * @brief     配置DHT11数据引脚为输出模式
 * @param     无
 * @return    无
 * @version   1.0
 * @note      配置PE0引脚为推挽输出模式，用于向DHT11发送信号
 */
void DHT11_Output()
{
	GPIO_InitTypeDef GPIO_InitStructure;
	
	/* 使能GPIOE端口时钟 */
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOE, ENABLE);
	
	/* 配置PE0引脚为输出模式 */
	GPIO_InitStructure.GPIO_Pin   = GPIO_Pin_0;        // PE0引脚
	GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_OUT;     // 输出模式
	GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;     // 推挽输出
	GPIO_InitStructure.GPIO_Speed = GPIO_Speed_100MHz; // 高速模式
	GPIO_InitStructure.GPIO_PuPd  = GPIO_PuPd_NOPULL;  // 无上下拉
	GPIO_Init(GPIOE, &GPIO_InitStructure);
	
	GPIO_SetBits(GPIOE, GPIO_Pin_0); // 默认高电平
}

/**
 * @name      DHT11_Input
 * @brief     配置DHT11数据引脚为输入模式
 * @param     无
 * @return    无
 * @version   1.0
 * @note      配置PE0引脚为输入模式，用于读取DHT11数据
 */
void DHT11_Input()
{
	GPIO_InitTypeDef GPIO_InitStructure;
	
	/* 使能GPIOE端口时钟 */
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOE, ENABLE);
	
	/* 配置PE0引脚为输入模式 */
	GPIO_InitStructure.GPIO_Pin   = GPIO_Pin_0;  // PE0引脚
	GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_IN; // 输入模式
	GPIO_Init(GPIOE, &GPIO_InitStructure);
}

/**
 * @name      DHT11_SendStart
 * @brief     向DHT11发送开始信号
 * @param     无
 * @return    无
 * @version   1.0
 * @note      发送DHT11通信起始信号：拉低20ms，然后拉高30us
 */
void DHT11_SendStart()
{
	DHT11_Output();  // 切换为输出模式
	
	/* 拉低电平并持续20ms */
	GPIO_ResetBits(GPIOE, GPIO_Pin_0);
	delay_ms(20);
	
	/* 拉高电平并持续30us */
	GPIO_SetBits(GPIOE, GPIO_Pin_0);
	delay_us(30);
}

/**
 * @name      DHT11_Config
 * @brief     DHT11初始化配置
 * @param     无
 * @return    无
 * @version   1.0
 * @note      初始化DHT11数据引脚为输出模式
 */
void DHT11_Config()
{
	DHT11_Output();  // 初始化为输出模式
}

/**
 * @name      DHT11_IsAck
 * @brief     检测DHT11响应信号
 * @param     无
 * @return    bool: true-响应成功, false-响应失败
 * @version   1.0
 * @note      检测DHT11是否响应MCU的起始信号
 */
bool DHT11_IsAck()
{
	uint32_t cnt = 0;
	
	/* 切换为输入模式准备接收响应 */
	DHT11_Input();
	
	/* 等待DHT11拉低电平（响应信号） */
	while(GPIO_ReadInputDataBit(GPIOE, GPIO_Pin_0) && cnt < 100)
	{
		delay_us(1);
		cnt++;
	}
	if(cnt >= 100)
		return false;  // 超时未响应
		
	cnt = 0;
	
	/* 等待DHT11拉高电平（结束响应信号） */
	while(!GPIO_ReadInputDataBit(GPIOE, GPIO_Pin_0) && cnt < 100)
	{
		delay_us(1);
		cnt++;
	}
	if(cnt >= 100)
		return false;  // 响应信号异常
	else
		return true;   // 响应成功
}

/**
 * @name      DHT11_ReadBit
 * @brief     读取DHT11发送的1位数据
 * @param     无
 * @return    uint8_t: 读取的位值(0或1)
 * @version   1.0
 * @note      通过检测高电平持续时间判断位值(26-28us为0, 70us为1)
 */
uint8_t DHT11_ReadBit()
{
	// 等待低电平开始（每个位都以50us低电平开始）
	while(GPIO_ReadInputDataBit(GPIOE, GPIO_Pin_0));
	
	// 等待低电平结束，开始高电平
	while(!GPIO_ReadInputDataBit(GPIOE, GPIO_Pin_0));
	
	// 延时40us后检测电平状态
	delay_us(40);
	if(GPIO_ReadInputDataBit(GPIOE, GPIO_Pin_0))
		return 1;  // 高电平持续70us，表示1
	else
		return 0;  // 高电平持续26-28us，表示0
}

/**
 * @name      DHT11_ReadByte
 * @brief     读取DHT11发送的1字节数据
 * @param     无
 * @return    uint8_t: 读取的字节数据
 * @version   1.0
 * @note      连续读取8位数据组成1字节（MSB先传）
 */
uint8_t DHT11_ReadByte()
{
	uint8_t data = 0;

	// 循环8次，接收一个字节（高位先传）
	for(uint8_t i = 0; i < 8; i++)
	{
		data <<= 1;        // 左移一位为新数据腾出空间
		data |= DHT11_ReadBit();  // 读取当前位并合并到数据中
	}
	return data;
}

/**
 * @name      DHT11_ReadData
 * @brief     读取DHT11温湿度数据
 * @param     data: 数据存储数组(5字节)
 * @return    bool: true-读取成功, false-读取失败
 * @version   1.0
 * @note      读取完整的40位数据并校验，数据格式：
 *            [0]:湿度整数部分, [1]:湿度小数部分
 *            [2]:温度整数部分, [3]:温度小数部分
 *            [4]:校验和(前4字节和)
 */
bool DHT11_ReadData(uint8_t data[5])
{
	uint8_t i = 0;
	
	// 1. MCU发送开始信号
	DHT11_SendStart();
	
	// 2. 检测DHT11响应
	if(DHT11_IsAck())
	{
		// 3. 读取5字节数据
		for(i = 0; i < 5; i++)
		{
			data[i] = DHT11_ReadByte();
		}
		
		// 4. 校验数据正确性
		if(data[4] == data[0] + data[1] + data[2] + data[3])
		{
			return true;  // 校验通过，数据正确
		}
		else
		{
			return false; // 校验失败，数据错误
		}
	}
	else
	{
		return false; // DHT11未响应
	}
}

/**
 * @name      main
 * @brief     主函数
 * @param     无
 * @return    int: 程序执行状态
 * @version   1.0
 * @note      DHT11温湿度传感器数据采集主程序
 */
int main()
{
	uint8_t Tdata[5] = {0};  // DHT11数据存储数组
	uint8_t buff[128] = {0}; // 串口输出缓冲区
	
	/* USART1初始化，波特率9600 */
	USART1_Config(9600);
	
	/* DHT11初始化 */
	DHT11_Config();
	
	/* 主循环 - 持续读取并显示温湿度数据 */
	while(1)
	{
		/* 读取DHT11温湿度数据 */
		if(DHT11_ReadData(Tdata))
		{
			/* 格式化温湿度数据 */
			sprintf((char *)buff, "温度：%d ℃, 湿度：%d %%RH\r\n", Tdata[2], Tdata[0]);
			
			/* 通过串口发送数据 */
			USART1_SendString((char *)buff);
		}
		else
		{
			/* 读取失败提示 */
			USART1_SendString("DHT11读取失败\r\n");
		}
		
		/* 延时500ms */
		delay_ms(500);
	}
}
