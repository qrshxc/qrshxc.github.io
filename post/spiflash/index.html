一、串行外设接口(SPI)深度解析

单总线协议指的是只采用一根信道来进行数据传输，通信指的是双方（MCU 与传感器）通过一根信道进行数据交互，所以按照数据的传输方向，只能采用半双工通信方式，比较典型的传感器就是温度传感器（DS18B20、DHT11……）

1. SPI基本概念

串行外设接口（Serial Peripheral Interface）的简称也叫做SPI，是一种高速的、全双工同步通信的一种接口，串行外设接口一般是需要4 根线来进行通信（NSS、MISO、MOSI、SCK），但是如果打算实现单向通信（最少3 根线），就可以利用这种机制实现一对多或者一对一的通信。

2.引脚定义

SPI 总线采用的环形结构，利用的是主从模式（主机—->从机）进行数据的传输，由于是同步通信，所以在主机发送数据的同时也会收到从机发送的数据。

3.工作模式

注意：采用SPI 接口进行通信，通信双方提前约定好使用哪个工作模式，此时主机可以有4种工作模式可以选择，此时选择哪种模式需要由从器件决定。主机需要配合从机！

绝大多数的采用SPI 接口通信的传感器，一般都是支持模式0 或者模式3！！！！

4.数据格式

主机与从机在通信的过程中传输的数据时以bit 为单位（串行传输），所以数据格式就十分重要，主机的数据格式必须要根据从机的数据格式进行设置（MSB 或者LSB），大多数使用SPI 接口通信的传感器一般都是使用MSB 高位先出。

二、Flash 闪存芯片的使用

1.基本概念

由于STM32F407ZET6 这颗MCU 内部只有512KB 的Flash 闪存容量，这512KB 还需要用于固化应用程序，所以剩余的空间可能不足以让用户使用了，所以需要额外的拓展存储IC，必须具备掉电不丢失，所以GEC-M4 开发板中集成了一颗串行Flash 芯片，型号是W25Q128，容量是128Mbit，该存储IC 采用SPI 接口进行通信。

/**
 ******************************************************************************
 * @file    main.c 
 * @author  qrshxc@163.com
 * @version V1.0
 * @date    2025.08.03
 * @brief   W25Q128 Flash存储器驱动
 *          引脚定义：
 *          CS   -- PB14
 *          MISO -- PB4  
 *          MOSI -- PB5
 *          CLK  -- PB3
 ******************************************************************************
 **/

#include "stm32f4xx.h"
#include <stdio.h>
#include <stdbool.h>
#include <string.h>

/* W25Q128引脚控制宏定义 */
#define W25Q128_CS(n)    ((n) ? GPIO_SetBits(GPIOB, GPIO_Pin_14) : GPIO_ResetBits(GPIOB, GPIO_Pin_14))
#define W25Q128_SCK(n)   ((n) ? GPIO_SetBits(GPIOB, GPIO_Pin_3) : GPIO_ResetBits(GPIOB, GPIO_Pin_3))
#define W25Q128_MOSI(n)  ((n) ? GPIO_SetBits(GPIOB, GPIO_Pin_5) : GPIO_ResetBits(GPIOB, GPIO_Pin_5))
#define W25Q128_MISO     GPIO_ReadInputDataBit(GPIOB, GPIO_Pin_4)

/* W25Q128指令定义 */
#define W25X_WriteEnable        0x06
#define W25X_WriteDisable       0x04
#define W25X_ReadStatusReg1     0x05
#define W25X_ReadStatusReg2     0x35
#define W25X_PageProgram        0x02
#define W25X_SectorErase        0x20
#define W25X_BlockErase         0xD8
#define W25X_ChipErase          0xC7
#define W25X_ReadData           0x03
#define W25X_FastRead           0x0B
#define W25X_ManufactDeviceID   0x90
#define W25X_JedecDeviceID      0x9F
#define W25X_ReleasePowerDown   0xAB

/**
 * @name      delay_us
 * @brief     微秒级延时函数
 * @param     nus: 延时的微秒数
 * @return    无
 * @version   1.0
 * @note      使用SysTick定时器实现精确微秒延时
 */
void delay_us(u32 nus)
{
    SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
    SysTick->LOAD = nus * 21 - 1; // 计数次数，递减计数，0结束
    SysTick->VAL = 0;     // 清除当前数值寄存器
    SysTick->CTRL = 1;    // 打开系统嘀嗒定时器，使用参考时钟
    while ((SysTick->CTRL & 0x00010000) == 0); // 等待计时完成
    SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
}

/**
 * @name      delay_ms
 * @brief     毫秒级延时函数
 * @param     nus: 延时的毫秒数
 * @return    无
 * @version   1.0
 * @note      使用SysTick定时器实现精确毫秒延时
 */
void delay_ms(u32 nus)
{
    SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
    SysTick->LOAD = nus * 21 * 1000 - 1; // 计数次数，递减计数，0结束
    SysTick->VAL = 0;     // 清除当前数值寄存器
    SysTick->CTRL = 1;    // 打开系统嘀嗒定时器，使用参考时钟
    while ((SysTick->CTRL & 0x00010000) == 0); // 等待计时完成
    SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
}

/**
 * @name      USART1_Config
 * @brief     配置USART1串口
 * @param     baud: 波特率
 * @return    无
 * @version   1.0
 * @note      配置USART1为异步模式，8位数据位，1位停止位，无校验
 */
void USART1_Config(uint32_t baud)
{
    USART_InitTypeDef USART_InitStructure;
    GPIO_InitTypeDef GPIO_InitStructure;

    /* 打开GPIOA时钟 */    
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOA, ENABLE);

    /* 打开USART1时钟 */
    RCC_APB2PeriphClockCmd(RCC_APB2Periph_USART1, ENABLE);    
    
    /* 设置PA9和PA10的复用功能为USART1 */
    GPIO_PinAFConfig(GPIOA, GPIO_PinSource9, GPIO_AF_USART1);  // PA9 - USART1_TX
    GPIO_PinAFConfig(GPIOA, GPIO_PinSource10, GPIO_AF_USART1); // PA10 - USART1_RX
    
    /* 配置GPIO的复用功能+初始化 */
    GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;        // 复用功能模式
    GPIO_InitStructure.GPIO_Speed = GPIO_Speed_100MHz;  // 高速模式
    GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;      // 推挽输出
    GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;        // 上拉模式
    GPIO_InitStructure.GPIO_Pin = GPIO_Pin_9 | GPIO_Pin_10; // PA9(TX), PA10(RX)
    GPIO_Init(GPIOA, &GPIO_InitStructure);

    /* 配置USART1参数+初始化 */
    USART_InitStructure.USART_BaudRate = baud;          // 波特率
    USART_InitStructure.USART_WordLength = USART_WordLength_8b; // 8位数据位
    USART_InitStructure.USART_StopBits = USART_StopBits_1;     // 1位停止位
    USART_InitStructure.USART_Parity = USART_Parity_No;        // 无校验位
    USART_InitStructure.USART_HardwareFlowControl = USART_HardwareFlowControl_None; // 无硬件流控
    USART_InitStructure.USART_Mode = USART_Mode_Rx | USART_Mode_Tx; // 收发模式
    USART_Init(USART1, &USART_InitStructure);

    /* 打开USART1 */
    USART_Cmd(USART1, ENABLE);
}

/**
 * @name      USART1_SendString
 * @brief     通过USART1发送字符串
 * @param     str: 要发送的字符串
 * @return    无
 * @version   1.0
 * @note      循环发送字符串中的每个字符，等待发送完成
 */
void USART1_SendString(const char* str)
{
    while(*str)
    {
        USART_SendData(USART1, *str++);  // 发送字符并指针后移
        while(!USART_GetFlagStatus(USART1, USART_FLAG_TXE)); // 等待发送完成
    }
}

/**
 * @name      W25Q128_Config
 * @brief     配置W25Q128 Flash引脚
 * @param     无
 * @return    无
 * @version   1.0
 * @note      配置SPI引脚：CS(PB14), MOSI(PB5), MISO(PB4), CLK(PB3)
 */
void W25Q128_Config()
{
    GPIO_InitTypeDef GPIO_InitStructure;
    
    /* 使能GPIOB时钟 */
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOB, ENABLE);
    
    /* 配置SPI输出引脚：CS、MOSI、CLK */
    GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;     // 输出模式
    GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz; // 50MHz速度
    GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;    // 推挽输出
    GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;      // 上拉模式
    GPIO_InitStructure.GPIO_Pin = GPIO_Pin_3 | GPIO_Pin_5 | GPIO_Pin_14; // CLK, MOSI, CS
    GPIO_Init(GPIOB, &GPIO_InitStructure);

    /* 配置MISO输入引脚 */
    GPIO_InitStructure.GPIO_Pin = GPIO_Pin_4;         // MISO引脚
    GPIO_InitStructure.GPIO_Mode = GPIO_Mode_IN;      // 输入模式
    GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;      // 上拉模式
    GPIO_Init(GPIOB, &GPIO_InitStructure);
    
    /* 初始状态：片选高电平（不选中） */
    W25Q128_CS(1);
    W25Q128_SCK(0);  // 时钟初始低电平（模式0）
    W25Q128_MOSI(1); // MOSI初始高电平
}

/**
 * @name      W25Q128_SendByte
 * @brief     通过SPI发送一个字节并接收返回数据
 * @param     byte: 要发送的字节
 * @return    uint8_t: 接收到的字节
 * @version   1.0
 * @note      软件模拟SPI通信，模式0，MSB先出
 */
uint8_t W25Q128_SendByte(uint8_t byte)
{
    uint8_t data = 0;
    
    /* 循环8次，发送和接收一个字节 */
    for(int i = 0; i < 8; i++)
    {
        /* 设置MOSI数据位 */
        if(byte & 0x80)
            W25Q128_MOSI(1);
        else
            W25Q128_MOSI(0);
        byte <<= 1;
        
        /* 时钟上升沿，从机采样数据 */
        delay_us(2);
        W25Q128_SCK(1);
        delay_us(2);
        
        /* 读取MISO数据 */
        data <<= 1;
        if(W25Q128_MISO)
            data |= 0x01;
        
        /* 时钟下降沿 */
        delay_us(2);
        W25Q128_SCK(0);
        delay_us(2);
    }
    
    return data;
}

/**
 * @name      W25Q128_ReadDeviceID
 * @brief     读取W25Q128设备ID
 * @param     无
 * @return    uint8_t: 设备ID
 * @version   1.0
 * @note      读取制造商ID，W25Q128应为0xEF
 */
uint8_t W25Q128_ReadDeviceID(void)
{
    uint8_t id = 0;
    
    W25Q128_CS(0);  // 片选使能
    
    /* 发送读取设备ID指令 */
    W25Q128_SendByte(W25X_ManufactDeviceID);
    
    /* 发送3字节地址(0x000000) */
    W25Q128_SendByte(0x00);
    W25Q128_SendByte(0x00);
    W25Q128_SendByte(0x00);
    
    /* 读取制造商ID */
    id = W25Q128_SendByte(0xFF);
    
    W25Q128_CS(1);  // 片选禁用
    
    return id;
}

/**
 * @name      W25Q128_WriteEnable
 * @brief     W25Q128写使能
 * @param     无
 * @return    无
 * @version   1.0
 * @note      在执行写操作前必须使能写操作
 */
void W25Q128_WriteEnable(void)
{
    W25Q128_CS(0);  // 片选使能
    W25Q128_SendByte(W25X_WriteEnable); // 发送写使能指令
    W25Q128_CS(1);  // 片选禁用
}

/**
 * @name      W25Q128_WriteDisable
 * @brief     W25Q128写失能
 * @param     无
 * @return    无
 * @version   1.0
 * @note      写操作完成后可禁用写操作
 */
void W25Q128_WriteDisable(void)
{
    W25Q128_CS(0);  // 片选使能
    W25Q128_SendByte(W25X_WriteDisable); // 发送写失能指令
    W25Q128_CS(1);  // 片选禁用
}

/**
 * @name      W25Q128_ReadStatusReg
 * @brief     读取W25Q128状态寄存器
 * @param     无
 * @return    uint8_t: 状态寄存器值
 * @version   1.0
 * @note      读取状态寄存器1，检查忙标志位(BUSY, bit0)
 */
uint8_t W25Q128_ReadStatusReg(void)
{
    uint8_t status;
    
    W25Q128_CS(0);  // 片选使能
    W25Q128_SendByte(W25X_ReadStatusReg1); // 发送读状态寄存器指令
    status = W25Q128_SendByte(0xFF);       // 读取状态寄存器值
    W25Q128_CS(1);  // 片选禁用
    
    return status;
}

/**
 * @name      W25Q128_WaitBusy
 * @brief     等待W25Q128空闲
 * @param     无
 * @return    无
 * @version   1.0
 * @note      等待Flash完成当前操作，检查BUSY位
 */
void W25Q128_WaitBusy(void)
{
    /* 等待BUSY位为0（空闲状态） */
    while((W25Q128_ReadStatusReg() & 0x01) == 0x01)
    {
        delay_us(100); // 短暂延时避免频繁查询
    }
}

/**
 * @name      W25Q128_EraseSector
 * @brief     擦除W25Q128一个扇区
 * @param     SectorAddr: 扇区地址
 * @return    无
 * @version   1.0
 * @note      擦除4KB扇区，擦除时间较长
 */
void W25Q128_EraseSector(uint32_t SectorAddr)
{
    /* 写使能 */
    W25Q128_WriteEnable();
    
    /* 发送扇区擦除指令 */
    W25Q128_CS(0);  // 片选使能
    W25Q128_SendByte(W25X_SectorErase);           // 扇区擦除指令
    W25Q128_SendByte((SectorAddr >> 16) & 0xFF);  // 地址字节2
    W25Q128_SendByte((SectorAddr >> 8) & 0xFF);   // 地址字节1
    W25Q128_SendByte(SectorAddr & 0xFF);          // 地址字节0
    W25Q128_CS(1);  // 片选禁用
    
    /* 等待擦除完成 */
    W25Q128_WaitBusy();
    
    /* 写失能 */
    W25Q128_WriteDisable();
}

/**
 * @name      W25Q128_WritePage
 * @brief     向W25Q128写入一页数据
 * @param     pBuffer: 数据缓冲区
 * @param     WriteAddr: 写入地址
 * @param     NumByteToWrite: 写入字节数
 * @return    无
 * @version   1.0
 * @note      页编程操作，最大256字节
 */
void W25Q128_WritePage(uint8_t* pBuffer, uint32_t WriteAddr, uint16_t NumByteToWrite)
{
    /* 参数检查 */
    if(NumByteToWrite > 256)
    {
        return; // 页大小超过限制
    }
    
    /* 写使能 */
    W25Q128_WriteEnable();
    
    /* 发送页编程指令和数据 */
    W25Q128_CS(0);  // 片选使能
    W25Q128_SendByte(W25X_PageProgram);            // 页编程指令
    W25Q128_SendByte((WriteAddr >> 16) & 0xFF);    // 地址字节2
    W25Q128_SendByte((WriteAddr >> 8) & 0xFF);     // 地址字节1
    W25Q128_SendByte(WriteAddr & 0xFF);            // 地址字节0
    
    /* 写入数据 */
    while(NumByteToWrite--)
    {
        W25Q128_SendByte(*pBuffer);
        pBuffer++;
    }
    
    W25Q128_CS(1);  // 片选禁用
    
    /* 等待写入完成 */
    W25Q128_WaitBusy();
    
    /* 写失能 */
    W25Q128_WriteDisable();
}

/**
 * @name      W25Q128_ReadBuffer
 * @brief     从W25Q128读取数据
 * @param     pBuffer: 数据缓冲区
 * @param     ReadAddr: 读取地址
 * @param     NumByteToRead: 读取字节数
 * @return    无
 * @version   1.0
 * @note      从指定地址读取数据到缓冲区
 */
void W25Q128_ReadBuffer(uint8_t* pBuffer, uint32_t ReadAddr, uint16_t NumByteToRead)
{
    W25Q128_CS(0);  // 片选使能
    
    /* 发送读数据指令和地址 */
    W25Q128_SendByte(W25X_ReadData);               // 读数据指令
    W25Q128_SendByte((ReadAddr >> 16) & 0xFF);     // 地址字节2
    W25Q128_SendByte((ReadAddr >> 8) & 0xFF);      // 地址字节1
    W25Q128_SendByte(ReadAddr & 0xFF);             // 地址字节0
    
    /* 读取数据 */
    while(NumByteToRead--)
    {
        *pBuffer = W25Q128_SendByte(0xFF);  // 发送dummy字节接收数据
        pBuffer++;
    }
    
    W25Q128_CS(1);  // 片选禁用
}

/**
 * @name      W25Q128_Test
 * @brief     W25Q128功能测试
 * @param     无
 * @return    bool: 测试结果
 * @version   1.0
 * @note      完整的Flash读写测试流程
 */
bool W25Q128_Test(void)
{
    uint8_t write_data[] = "Hello W25Q128 Flash Test!";
    uint8_t read_data[50] = {0};
    uint8_t device_id;
    
    /* 读取设备ID验证连接 */
    device_id = W25Q128_ReadDeviceID();
    if(device_id != 0xEF)  // W25Q128制造商ID应为0xEF
    {
        return false; // 设备ID不匹配
    }
    
    /* 擦除第一个扇区 */
    W25Q128_EraseSector(0x000000);
    
    /* 写入测试数据 */
    W25Q128_WritePage(write_data, 0x000000, strlen((char*)write_data));
    
    /* 读取验证数据 */
    W25Q128_ReadBuffer(read_data, 0x000000, strlen((char*)write_data));
    
    /* 数据校验 */
    if(memcmp(write_data, read_data, strlen((char*)write_data)) != 0)
    {
        return false; // 数据校验失败
    }
    
    return true; // 测试通过
}

/**
 * @name      main
 * @brief     主函数
 * @param     无
 * @return    int: 程序执行状态
 * @version   1.0
 * @note      W25Q128 Flash存储器测试主程序
 */
int main()
{
    uint8_t write_buffer[128] = "你好，W25Q128 Flash测试！";
    uint8_t read_buffer[128] = {0};
    char info_buffer[64] = {0};
    
    /* 系统初始化 */
    SystemInit();
    
    /* USART1初始化，波特率9600 */
    USART1_Config(9600);
    
    /* W25Q128初始化 */
    W25Q128_Config();
    
    /* 设备ID检测 */
    uint8_t device_id = W25Q128_ReadDeviceID();
    sprintf(info_buffer, "W25Q128 Device ID: 0x%02X\r\n", device_id);
    USART1_SendString(info_buffer);
    
    /* Flash功能测试 */
    if(W25Q128_Test())
    {
        USART1_SendString("W25Q128 Test: PASS\r\n");
    }
    else
    {
        USART1_SendString("W25Q128 Test: FAIL\r\n");
        while(1); // 测试失败时停止
    }
    
    /* 演示数据读写 */
    USART1_SendString("Writing data to Flash...\r\n");
    W25Q128_EraseSector(0x001000);  // 使用第二个扇区避免冲突
    W25Q128_WritePage(write_buffer, 0x001000, strlen((char*)write_buffer));
    
    USART1_SendString("Reading data from Flash...\r\n");
    W25Q128_ReadBuffer(read_buffer, 0x001000, strlen((char*)write_buffer));
    
    USART1_SendString("Read Data: ");
    USART1_SendString((char*)read_buffer);
    USART1_SendString("\r\n");
    
    /* 主循环 */
    while(1)
    {
        delay_ms(1000);
    }
}
