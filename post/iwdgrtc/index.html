一、看门狗外设的原理与应用

随着单片机的发展，单片机在家用电器、工业自动化、生产过程控制、智能仪器仪表等领域的应用越来越广泛。然而处于同一电力系统中的各种电气设备通过电或磁的联系彼此紧密相连，相互影响，由于运行方式的改变、故障、开关操作等引起的电磁振荡会波及很多电气设备。这对我们单片机系统的可靠性与安全性构成了极大的威胁。单片机测控系统必须长期稳定、可靠运行，否则将导致控制误差加大，严重时会使系统失灵，甚至造成巨大损失。因此单片机的抗干扰能力已经成为不容忽视的问题。

由于单片机在运行的过程经常会受到外部电磁场的干扰，就可能会出现程序“跑飞”，比如出现数据的缺失、导致寄存器中的数据发生变化、程序指针PC 指向非法地址……..，为了提高程序的可靠性，就提供了看门狗外设，在程序出现由软件或者硬件导致的未知问题，利用 看门狗外设对芯片进行复位，相当于重新运行程序。

如果程序运行没有任何问题，则可以”按时喂狗”，就不会让芯片复位，一旦程序出现问题导致无法喂狗，则看门狗会自动复位程序。

使用看门狗的目的：为了提供程序的可靠性，防止由软件或者硬件造成的问题的出现。如果想要了解看门狗的资源和特点，可以参考STM32F4 中文参考手册。

1.分类说明

STM32F4 系列的芯片提供的两个看门狗，一个叫做独立看门狗，一个叫做窗口看门狗。两个看门狗的使用流程差不多，但是相比于独立看门狗，窗口看门狗的要求更多。

/**

 * @file    IWDG.c 
 * @author  qrshxc@163.com
 * @version V1.0

#include "stm32f4xx.h"

/**
 * @name      delay_ms
 * @brief     毫秒级延时函数
 * @param     nus: 延时的毫秒数
 * @return    无
 * @version   1.0
 * @note      使用SysTick定时器实现精确毫秒延时
 */
void delay_ms(u32 nus)
{
	SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
	SysTick->LOAD = nus * 21 * 1000 -1; // 计数次数，递减计数，0结束
	SysTick->VAL = 0;     // 清除当前数值寄存器
	SysTick->CTRL = 1;    // 打开系统嘀嗒定时器，使用参考时钟
	while ((SysTick->CTRL & 0x00010000)==0); // 等待计时完成
	SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器
}

/**
 * @name      USART1_Config
 * @brief     配置USART1串口
 * @param     baud: 波特率
 * @return    无
 * @version   1.0
 * @note      配置USART1为异步模式，8位数据位，1位停止位，无校验
 */
void USART1_Config(uint32_t baud)
{
	USART_InitTypeDef USART_InitStructure;
	GPIO_InitTypeDef GPIO_InitStructure;

	/* 打开GPIOA时钟 */	
	RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOA, ENABLE);

	/* 打开USART1时钟 */
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_USART1, ENABLE);	
	
	/* 设置PA9和PA10的复用功能为USART1 */
	GPIO_PinAFConfig(GPIOA, GPIO_PinSource9, GPIO_AF_USART1);  // PA9 - USART1_TX
	GPIO_PinAFConfig(GPIOA, GPIO_PinSource10, GPIO_AF_USART1); // PA10 - USART1_RX
	
	/* 配置GPIO的复用功能+初始化 */
	GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;        // 复用功能模式
	GPIO_InitStructure.GPIO_Speed = GPIO_Speed_100MHz;  // 高速模式
	GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;      // 推挽输出
	GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;        // 上拉模式
	GPIO_InitStructure.GPIO_Pin = GPIO_Pin_9 | GPIO_Pin_10; // PA9(TX), PA10(RX)
	GPIO_Init(GPIOA, &GPIO_InitStructure);

	/* 配置USART1参数+初始化 */
	USART_InitStructure.USART_BaudRate = baud;          // 波特率
	USART_InitStructure.USART_WordLength = USART_WordLength_8b; // 8位数据位
	USART_InitStructure.USART_StopBits = USART_StopBits_1;     // 1位停止位
	USART_InitStructure.USART_Parity = USART_Parity_No;        // 无校验位
	USART_InitStructure.USART_HardwareFlowControl = USART_HardwareFlowControl_None; // 无硬件流控
	USART_InitStructure.USART_Mode = USART_Mode_Rx | USART_Mode_Tx; // 收发模式
	USART_Init(USART1, &USART_InitStructure);

	/* 打开USART1 */
	USART_Cmd(USART1, ENABLE);
}

/**
 * @name      USART1_SendString
 * @brief     通过USART1发送字符串
 * @param     str: 要发送的字符串
 * @return    无
 * @version   1.0
 * @note      循环发送字符串中的每个字符，等待发送完成
 */
void USART1_SendString(const char* str)
{
	while(*str)
	{
		USART_SendData(USART1, *str++);  // 发送字符并指针后移
		while(!USART_GetFlagStatus(USART1, USART_FLAG_TXE)); // 等待发送完成
	}
}

/**
 * @name      IWDOG_Config
 * @brief     配置独立看门狗
 * @param     无
 * @return    无
 * @version   1.0
 * @note      配置IWDG超时时间为20ms
 */
void IWDOG_Config()
{
	/* 使能对IWDG寄存器的写访问 */
	IWDG_WriteAccessCmd(IWDG_WriteAccess_Enable);
	
	/* 设置看门狗预分频值 */
	/* LSI 32kHz / 32 = 1kHz, 计数周期 = 1ms */
	IWDG_SetPrescaler(IWDG_Prescaler_32);
	
	/* 设置重载值: 20次 * 1ms = 20ms超时时间 */ 
	IWDG_SetReload(20);
 
	/* 喂狗操作 - 重置计数器 */
	IWDG_ReloadCounter();
 
	/* 使能独立看门狗 */
	IWDG_Enable();
}

/**
 * @name      main
 * @brief     主函数
 * @param     无
 * @return    int: 程序执行状态
 * @version   1.0
 * @note      独立看门狗应用演示，检测复位源并定期喂狗
 */
int main()
{
	/* USART1初始化，波特率9600 */
	USART1_Config(9600);
	
	/* 检查是否IWDG复位 */
	if (RCC_GetFlagStatus(RCC_FLAG_IWDGRST) != RESET)
	{
		/* 看门狗复位 */
		USART1_SendString("System Reset by IWDG\r\n");
		RCC_ClearFlag();  // 清除复位标志
	}
	else
	{
		/* 用户复位 */ 
		USART1_SendString("System Reset by User\r\n");
		RCC_ClearFlag();  // 清除复位标志
	}
	
	/* 配置独立看门狗，超时时间20ms */
	IWDOG_Config();
	
	/* 主循环 */
	while(1)
	{
		/* 喂狗操作 - 重置看门狗计数器 */
		IWDG_ReloadCounter();
		
		/* 延时19ms，确保在20ms超时前喂狗 */
		delay_ms(19);
	}
}

/**
 ******************************************************************************
 * @file    RTC.c 
 * @author  qrshxc@163.com
 * @version V1.0
 * @date    2025.7.30
 * @brief   RTC实时时钟显示日期和时间
 ******************************************************************************
 **/

#include "stm32f4xx.h"
#include <stdio.h>

/* 全局变量定义 */
__IO uint32_t uwTimeDisplay = 0;           // 时间显示标志
RTC_InitTypeDef RTC_InitStructure;         // RTC初始化结构体
RTC_TimeTypeDef RTC_TimeStructure;         // RTC时间结构体
RTC_DateTypeDef RTC_DateStructure;         // RTC日期结构体
uint8_t aShowTime[50] = {0};               // 时间显示缓冲区
uint8_t aShowDate[50] = {0};               // 日期显示缓冲区

/**
 * @name      delay_ms
 * @brief     毫秒级延时函数
 * @param     nus: 延时的毫秒数
 * @return    无
 * @version   1.0
 * @note      使用SysTick定时器实现精确毫秒延时

{

	SysTick->CTRL = 0;    // 关闭系统嘀嗒定时器

/**
 * @name      USART1_Config
 * @brief     配置USART1串口
 * @param     baud: 波特率
 * @return    无
 * @version   1.0
 * @note      配置USART1为异步模式，8位数据位，1位停止位，无校验
 */

{
	USART_InitTypeDef USART_InitStructure;
	GPIO_InitTypeDef GPIO_InitStructure;

	/* 打开GPIOA时钟 */	
	RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOA, ENABLE);

	GPIO_InitStructure.GPIO_Speed = GPIO_Speed_100MHz;  // 高速模式
	GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;      // 推挽输出
	GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;        // 上拉模式
	GPIO_InitStructure.GPIO_Pin = GPIO_Pin_9 | GPIO_Pin_10; // PA9(TX), PA10(RX)

	USART_InitStructure.USART_Parity = USART_Parity_No;        // 无校验位
	USART_InitStructure.USART_HardwareFlowControl = USART_HardwareFlowControl_None; // 无硬件流控
	USART_InitStructure.USART_Mode = USART_Mode_Rx | USART_Mode_Tx; // 收发模式

/**
 * @name      RTC_WKUP_IRQHandler
 * @brief     RTC唤醒中断服务函数
 * @param     无
 * @return    无
 * @version   1.0
 * @note      处理RTC唤醒中断，设置时间显示标志
 */
void RTC_WKUP_IRQHandler(void)
{
	if(RTC_GetITStatus(RTC_IT_WUT) != RESET)
	{
		uwTimeDisplay = 1;  // 设置时间显示标志
		RTC_ClearITPendingBit(RTC_IT_WUT);  // 清除RTC中断标志
		EXTI_ClearITPendingBit(EXTI_Line22); // 清除外部中断标志
	}
}

/**
 * @name      RTC_Config
 * @brief     配置RTC实时时钟
 * @param     无
 * @return    无
 * @version   1.0
 * @note      配置RTC使用LSE时钟源，设置初始时间和日期
 */
void RTC_Config()
{
	NVIC_InitTypeDef NVIC_InitStructure;
	EXTI_InitTypeDef EXTI_InitStructure;
	
	/* 打开PWR电源控制时钟 */
	RCC_APB1PeriphClockCmd(RCC_APB1Periph_PWR, ENABLE);

	/* 允许访问备份域寄存器 */
	PWR_BackupAccessCmd(ENABLE);
	
	/* 配置RTC唤醒中断 */
	NVIC_InitStructure.NVIC_IRQChannel = RTC_WKUP_IRQn;
	NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = 1;
	NVIC_InitStructure.NVIC_IRQChannelSubPriority = 0;
	NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
	NVIC_Init(&NVIC_InitStructure);

	/* 配置外部中断线22（RTC唤醒） */
	EXTI_ClearITPendingBit(EXTI_Line22);
	EXTI_InitStructure.EXTI_Line = EXTI_Line22;
	EXTI_InitStructure.EXTI_Mode = EXTI_Mode_Interrupt;
	EXTI_InitStructure.EXTI_Trigger = EXTI_Trigger_Rising;
	EXTI_InitStructure.EXTI_LineCmd = ENABLE;
	EXTI_Init(&EXTI_InitStructure);

	/* 打开LSE低速外部晶振（32.768kHz） */
	RCC_LSEConfig(RCC_LSE_ON);

	/* 等待LSE振荡器稳定 */
	while(RCC_GetFlagStatus(RCC_FLAG_LSERDY) == RESET)
	{
	}
	
	/* 选择LSE作为RTC时钟源 */
	RCC_RTCCLKConfig(RCC_RTCCLKSource_LSE);

	/* 使能RTC时钟 */
	RCC_RTCCLKCmd(ENABLE);

	/* 等待RTC寄存器同步 */
	RTC_WaitForSynchro();
	
	/* 设置初始时间：16:00:40 */
	RTC_TimeStructure.RTC_Hours   = 0x16;  // 16小时
	RTC_TimeStructure.RTC_Minutes = 0x00;  // 00分钟
	RTC_TimeStructure.RTC_Seconds = 0x40;  // 40秒
	
	/* 设置初始日期：2025年10月5日 星期日 */
	RTC_DateStructure.RTC_Month = 0x07;    // 7月
	RTC_DateStructure.RTC_Date = 0x30;     // 30日
	RTC_DateStructure.RTC_Year = 0x25;     // 25年（2025）
	RTC_DateStructure.RTC_WeekDay = 0x03;  // 星期三
	
	/* 配置RTC日历参数 */
	RTC_InitStructure.RTC_AsynchPrediv = 0x7F - 1;  // 异步预分频值
	RTC_InitStructure.RTC_SynchPrediv =  0xFF - 1;  // 同步预分频值
	RTC_InitStructure.RTC_HourFormat = RTC_HourFormat_24; // 24小时制
	RTC_Init(&RTC_InitStructure);
	
	/* 设置RTC时间和日期（BCD格式） */
	RTC_SetTime(RTC_Format_BCD, &RTC_TimeStructure);  
	RTC_SetDate(RTC_Format_BCD, &RTC_DateStructure); 
	
	/* 配置RTC唤醒时钟和计数器 */
	RTC_WakeUpClockConfig(RTC_WakeUpClock_CK_SPRE_16bits);
	RTC_SetWakeUpCounter(1-1);  // 设置唤醒计数器
	
	/* 使能RTC唤醒中断 */
	RTC_ITConfig(RTC_IT_WUT, ENABLE);
	RTC_ClearITPendingBit(RTC_IT_WUT);

	/* 启动RTC唤醒功能 */
	RTC_WakeUpCmd(ENABLE); 
}

/**
 * @name      Calendar_Show
 * @brief     显示日历信息
 * @param     无
 * @return    无
 * @version   1.0
 * @note      获取并显示当前时间和日期信息
 */
static void Calendar_Show(void)
{
	/* 检查时间显示标志 */
	if (uwTimeDisplay == 1)
	{
		/* 获取当前时间（二进制格式） */
		RTC_GetTime(RTC_Format_BIN, &RTC_TimeStructure);
		
		/* 获取当前日期（二进制格式） */
		RTC_GetDate(RTC_Format_BIN, &RTC_DateStructure);
		
		/* 格式化时间字符串：hh:mm:ss */
		sprintf((char*)aShowTime, "%0.2d:%0.2d:%0.2d\r\n",
				RTC_TimeStructure.RTC_Hours, 
				RTC_TimeStructure.RTC_Minutes, 
				RTC_TimeStructure.RTC_Seconds);
		
		/* 格式化日期字符串：yyyy-mm-dd */
		sprintf((char*)aShowDate, "20%0.2d-%0.2d-%0.2d\r\n",
				RTC_DateStructure.RTC_Year, 
				RTC_DateStructure.RTC_Month, 
				RTC_DateStructure.RTC_Date);
		
		/* 通过串口发送时间和日期信息 */
		USART1_SendString((char*)aShowTime);
		USART1_SendString((char*)aShowDate);
		
		/* 清除时间显示标志 */
		uwTimeDisplay = 0;
	}
}

/**
 * @name      main
 * @brief     主函数
 * @param     无
 * @return    int: 程序执行状态
 * @version   1.0
 * @note      RTC实时时钟主程序，周期性显示时间和日期
 */
int main()
{
	/* USART1初始化，波特率9600 */
	USART1_Config(9600);
	
	/* RTC实时时钟初始化 */
	RTC_Config();
	
	/* 主循环 */
	while(1)
	{
		/* 显示日历信息 */
		Calendar_Show();
		
		/* 延时500ms */
		delay_ms(500);
	}
}
