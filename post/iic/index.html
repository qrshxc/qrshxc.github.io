一、内部集成电路概述

1. 基本概念

内部集成电路（Inter Integrated Circuit）的简称叫做IIC 或者I2C，是一种简单的、半双工同步通信的串行通信接口，IIC 总线是上世纪80 年代（1982 年）由飞利浦公司设计出来，当时的目的是为了给MCU 和外围芯片提供更简单的交互方式。

2.引脚说明

IIC 总线只需要两根引脚就可以实现通信，一根是数据线（SDA Serial Data），另一根是时钟线（SCL Serial Clock），所有通过IIC 接口通信的外围器件都挂载在IIC 总线上，通过这种机制就可以实现多机通信。

可以看到，外围器件的时钟线和数据线都是挂载在IIC 总线（由主控芯片提供），并且在空闲状态下所有器件的时钟线（SCL）和数据线（SDA）都被总线的上拉电阻拉高，这样就可以把SDA 引脚和SCL 引脚设置为开漏模式即可，好处是防止短路。

思考：如果IIC 总线上挂载了多个外围器件，如何与某一个器件进行单独通信？ 答：器件地址

3.通信速率

可以看到IIC 总线支持不同的通信速率，但是一般常用的标准速率100KHZ，但是有的外围器件可以支持高达400KHZ 的通信速率，而由于IIC 总线是半双工通信，所以同一时刻只能接收或者发送，也就是说IIC 总线一般是为了控制，不适合作为大量数据传输的接口。

/**
******************************************************************************
* @file   myiic.c 
* @author  
* @version V1.0
* @date    2025.08.05
* @brief   IIC驱动源文件
            SCL--PB8
            SDA--PB9
******************************************************************************
**/

#include "myiic.h"
#include "delay.h"

/**
  * @function IIC_SCLConfig 
  * @brief    SCL引脚配置
  * @param    None
  * @retval   None
  */
void IIC_SCLConfig(void)
{
    GPIO_InitTypeDef GPIO_InitStructure;

    // 打开GPIOB端口时钟
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOB, ENABLE);

    // SCL引脚配置(PB8)
    GPIO_InitStructure.GPIO_Mode   = GPIO_Mode_OUT;    // 输出模式
    GPIO_InitStructure.GPIO_Speed  = GPIO_Speed_50MHz; // 输出速度50MHz
    GPIO_InitStructure.GPIO_OType  = GPIO_OType_PP;    // 推挽输出
    GPIO_InitStructure.GPIO_PuPd   = GPIO_PuPd_UP;     // 上拉电阻
    GPIO_InitStructure.GPIO_Pin    = GPIO_Pin_8;       // PB8引脚
    GPIO_Init(GPIOB, &GPIO_InitStructure);
}

/**
  * @function IIC_SDAConfig 
  * @brief    SDA引脚配置
  * @param    GPIO_Mode: 引脚模式（输入或输出）
  * @retval   None
  */
void IIC_SDAConfig(GPIOMode_TypeDef GPIO_Mode)
{
    GPIO_InitTypeDef GPIO_InitStructure;

    // 打开GPIOB端口时钟
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOB, ENABLE);

    // SDA引脚配置(PB9)
    GPIO_InitStructure.GPIO_Mode   = GPIO_Mode;        // 输入或输出模式
    GPIO_InitStructure.GPIO_Speed  = GPIO_Speed_50MHz; // 输出速度50MHz
    GPIO_InitStructure.GPIO_OType  = GPIO_OType_PP;    // 推挽输出
    GPIO_InitStructure.GPIO_PuPd   = GPIO_PuPd_UP;     // 上拉电阻
    GPIO_InitStructure.GPIO_Pin    = GPIO_Pin_9;       // PB9引脚
    GPIO_Init(GPIOB, &GPIO_InitStructure);
}

/**
  * @function IIC_Config 
  * @brief    IIC总线初始化
  * @param    None
  * @retval   None
  */
void IIC_Config(void)
{
    // 1.设置SDA和SCL为输出模式
    IIC_SCLConfig();
    IIC_SDAConfig(GPIO_Mode_OUT);

    // 2.确保SDA和SCL处于空闲状态（高电平）
    SDA_SET(1);
    SCL_SET(1);
    delay_us(5);
}

/**
  * @function IIC_Start 
  * @brief    IIC起始信号
  * @param    None
  * @retval   None
  */
void IIC_Start(void)
{
    // 1.设置SDA引脚为输出模式
    IIC_SDAConfig(GPIO_Mode_OUT);

    // 2.确保SDA和SCL处于空闲状态
    SDA_SET(1);
    SCL_SET(1);
    delay_us(5);

    // 3.在SCL为高电平时，SDA从高变低，产生起始信号
    SDA_SET(0);
    delay_us(5);

    // 4.拉低SCL，准备数据传输
    SCL_SET(0);
}

/**
  * @function IIC_SendByte 
  * @brief    IIC发送一个字节
  * @param    Byte: 要发送的字节数据
  * @retval   None
  */
void IIC_SendByte(uint8_t Byte)
{
    uint8_t i = 0;
    
    // 1.设置SDA引脚为输出模式
    IIC_SDAConfig(GPIO_Mode_OUT);

    // 2.确保SCL为低电平，主机准备数据
    SCL_SET(0);
    delay_us(5);

    // 3.循环发送8位数据，高位先出(MSB)
    for(i = 0; i < 8; i++)
    {
        // 4.根据字节的最高位设置SDA电平
        if(Byte & 0x80)
            SDA_SET(1);  // 最高位为1，SDA置高
        else
            SDA_SET(0);  // 最高位为0，SDA置低

        Byte <<= 1;      // 左移一位，准备发送下一位
        delay_us(5);

        // 5.拉高SCL，从机读取数据位
        SCL_SET(1);
        delay_us(5);

        // 6.拉低SCL，主机准备下一个数据位
        SCL_SET(0);
        delay_us(5);
    }
}

/**
  * @function IIC_IsSlaveACK 
  * @brief    检测从机应答信号
  * @param    None
  * @retval   bool: true-从机应答, false-从机未应答
  */
bool IIC_IsSlaveACK(void)
{
    bool ack;
    
    // 1.设置SDA引脚为输入模式，准备读取从机应答
    IIC_SDAConfig(GPIO_Mode_IN);

    // 2.拉低SCL，从机准备应答位
    SCL_SET(0);
    delay_us(5);

    // 3.拉高SCL，主机读取应答状态
    SCL_SET(1);
    delay_us(5);

    // 4.读取SDA状态：1表示未应答，0表示已应答
    if(SDA_READ)
        ack = false;  // 未应答(NACK)
    else
        ack = true;   // 已应答(ACK)
    
    // 5.拉低SCL，结束应答周期
    SCL_SET(0);
    delay_us(5);
    
    return ack;
}

/**
  * @function IIC_ReadByte 
  * @brief    IIC读取一个字节
  * @param    None
  * @retval   uint8_t: 读取到的字节数据
  */
uint8_t IIC_ReadByte(void)
{
    uint8_t i = 0, data = 0;

    // 1.设置SDA引脚为输入模式，准备读取数据
    IIC_SDAConfig(GPIO_Mode_IN);

    // 2.拉低SCL，从机准备数据
    SCL_SET(0);
    delay_us(5);

    // 3.循环读取8位数据，高位先出(MSB)
    for(i = 0; i < 8; i++)
    {
        // 4.拉高SCL，主机读取数据位
        SCL_SET(1);
        delay_us(5);

        // 5.读取SDA状态并组合成字节
        data <<= 1;          // 左移一位，为新的数据位腾出空间
        data |= SDA_READ;    // 读取当前数据位
        
        // 6.拉低SCL，从机准备下一个数据位
        SCL_SET(0);
        delay_us(5);
    }

    // 7.返回读取到的完整字节
    return data;
}

/**
  * @function IIC_MasterACK 
  * @brief    主机发送应答信号
  * @param    ack: 应答标志（1-非应答，0-应答）
  * @retval   None
  */
void IIC_MasterACK(uint8_t ack)
{
    // 1.设置SDA引脚为输出模式，准备发送应答
    IIC_SDAConfig(GPIO_Mode_OUT);

    // 2.拉低SCL，主机准备应答位
    SCL_SET(0);
    delay_us(5);

    // 3.根据ack参数设置SDA电平
    if(ack)
        SDA_SET(1);  // 非应答(NACK)
    else
        SDA_SET(0);  // 应答(ACK)

    delay_us(5);

    // 4.拉高SCL，从机读取应答位
    SCL_SET(1);
    delay_us(5);
    
    // 5.拉低SCL，结束应答周期
    SCL_SET(0);
    delay_us(5);
}

/**
  * @function IIC_Stop 
  * @brief    IIC停止信号
  * @param    None
  * @retval   None
  */
void IIC_Stop(void)
{
    // 1.设置SDA引脚为输出模式
    IIC_SDAConfig(GPIO_Mode_OUT);

    // 2.确保SDA和SCL都为低电平
    SDA_SET(0);
    SCL_SET(0);
    delay_us(5);

    // 3.先拉高SCL
    SCL_SET(1);
    delay_us(5);

    // 4.在SCL为高电平时，SDA从低变高，产生停止信号
    SDA_SET(1);
    delay_us(5);
}

/**

#include "at24c02.h"
#include "delay.h"

/**
  * @function AT24C02_Config 
  * @brief    AT24C02初始化
  * @param    None
  * @retval   None
  */
void AT24C02_Config(void)
{
    // 初始化IIC总线
    IIC_Config();
}

/**
  * @function AT24C02_ByteWrite 
  * @brief    AT24C02字节写入操作
  * @param    Addr: 存储地址(0x00-0xFF)
  * @param    Byte: 要写入的字节数据
  * @retval   bool: true-写入成功, false-写入失败
  */
bool AT24C02_ByteWrite(uint8_t Addr, uint8_t Byte)
{
    // 1.开始信号
    IIC_Start();
    
    // 2.发送器件地址(写模式)
    IIC_SendByte(AT24C02_WRITE_ADDR);
    
    // 3.检测从机应答
    if(IIC_IsSlaveACK() == false) 
    {
        IIC_Stop();
        return false;  // 器件地址错误
    }
    
    // 4.发送存储地址
    IIC_SendByte(Addr);
    
    // 5.检测从机应答
    if(IIC_IsSlaveACK() == false) 
    {
        IIC_Stop();
        return false;  // 存储地址错误
    }
    
    // 6.发送数据字节
    IIC_SendByte(Byte);
    
    // 7.检测从机应答
    if(IIC_IsSlaveACK() == false) 
    {
        IIC_Stop();
        return false;  // 数据写入错误
    }
    
    // 8.停止信号
    IIC_Stop();
    
    // 9.延时5ms，等待EEPROM内部写操作完成
    delay_ms(5);
    
    return true;
}

/**
  * @function AT24C02_CurrentAddressRead 
  * @brief    AT24C02当前地址读取
  * @param    None
  * @retval   uint8_t: 读取到的字节数据
  */
uint8_t AT24C02_CurrentAddressRead(void)
{
    uint8_t recv = 0;
    
    // 1.开始信号
    IIC_Start();
    
    // 2.发送器件地址(读模式)
    IIC_SendByte(AT24C02_READ_ADDR);
    
    // 3.检测从机应答
    if(IIC_IsSlaveACK() == false) 
    {
        IIC_Stop();
        return 0xFF;  // 读取失败，返回0xFF
    }
    
    // 4.读取字节数据
    recv = IIC_ReadByte();
    
    // 5.发送非应答信号(表示读取结束)
    IIC_MasterACK(0x01);
    
    // 6.停止信号
    IIC_Stop();
    
    return recv;
}

/**
  * @function AT24C02_RandomAddressRead 
  * @brief    AT24C02随机地址读取
  * @param    DestAddr: 目标读取地址(0x00-0xFF)
  * @retval   uint8_t: 读取到的字节数据
  */
uint8_t AT24C02_RandomAddressRead(uint8_t DestAddr)
{
    uint8_t recv = 0;
    
    // 1.开始信号
    IIC_Start();
    
    // 2.发送器件地址(写模式)，用于指定读取地址
    IIC_SendByte(AT24C02_WRITE_ADDR);
    
    // 3.检测从机应答
    if(IIC_IsSlaveACK() == false) 
    {
        IIC_Stop();
        return 0xFF;  // 器件地址错误
    }
    
    // 4.发送要读取的存储地址
    IIC_SendByte(DestAddr);
    
    // 5.检测从机应答
    if(IIC_IsSlaveACK() == false) 
    {
        IIC_Stop();
        return 0xFF;  // 存储地址错误
    }
    
    // 6.重新开始信号(用于切换为读模式)
    IIC_Start();
    
    // 7.发送器件地址(读模式)
    IIC_SendByte(AT24C02_READ_ADDR);
    
    // 8.检测从机应答
    if(IIC_IsSlaveACK() == false) 
    {
        IIC_Stop();
        return 0xFF;  // 读取模式切换失败
    }
    
    // 9.读取字节数据
    recv = IIC_ReadByte();
    
    // 10.发送非应答信号(表示读取结束)
    IIC_MasterACK(0x01);
    
    // 11.停止信号
    IIC_Stop();
    
    return recv;
}

/**
  * @function SCL_Config 
  * @brief    SCL引脚配置
  * @param    None
  * @retval   None
  */
void SCL_Config(void)
{
    GPIO_InitTypeDef GPIO_InitStructure;
    
    /* 使能GPIOG端口时钟 */
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOG, ENABLE);
    
    /* SCL引脚配置(PG7) */
    GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;      // 输出模式
    GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;  // 输出速度50MHz
    GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;     // 推挽输出
    GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;       // 上拉电阻
    GPIO_InitStructure.GPIO_Pin = GPIO_Pin_7;          // PG7引脚
    GPIO_Init(GPIOG, &GPIO_InitStructure);
}

/**
  * @function IIC_Config 
  * @brief    IIC总线初始化
  * @param    None
  * @retval   None
  */
void IIC_Config(void)
{
    /* 初始化SCL和SDA引脚 */
    SCL_Config();
    SDA_Config(GPIO_Mode_OUT);
    
    /* 设置SDA和SCL为高电平（空闲状态） */
    SDA_SET(1);
    SCL_SET(1);
    delay_us(5);
}

/**
  * @function IIC_Start 
  * @brief    IIC起始信号
  * @param    None
  * @retval   None
  */
void IIC_Start(void)
{
    /* 设置SDA为输出模式 */
    SDA_Config(GPIO_Mode_OUT);
    
    /* 确保总线空闲状态 */
    SDA_SET(1);
    SCL_SET(1);
    delay_us(5);

    /* 在SCL为高时，SDA从高变低产生起始信号 */
    SDA_SET(0);
    delay_us(5);

    /* 拉低SCL，准备数据传输 */
    SCL_SET(0);
    delay_us(5);
}

/**
  * @function IIC_SendByte 
  * @brief    IIC发送一个字节
  * @param    byte: 要发送的字节数据
  * @retval   None
  */
void IIC_SendByte(uint8_t byte)
{
    uint8_t i = 0;
    
    /* 设置SDA为输出模式 */
    SDA_Config(GPIO_Mode_OUT);

    /* 拉低SCL，主机准备数据 */
    SCL_SET(0);
    delay_us(5);

    /* 循环发送8位数据，高位先出(MSB) */
    for(i = 0; i < 8; i++)
    {
        /* 根据字节的最高位设置SDA电平 */
        if(byte & 0x80)
            SDA_SET(1);  // 最高位为1，SDA置高
        else
            SDA_SET(0);  // 最高位为0，SDA置低
            
        byte <<= 1;      // 左移一位，准备发送下一位
        delay_us(5);

        /* 拉高SCL，从机读取数据位 */
        SCL_SET(1);
        delay_us(5);

        /* 拉低SCL，主机准备下一个数据位 */
        SCL_SET(0);
        delay_us(5);
    }
}

/**
  * @function IIC_IsSlaveAck 
  * @brief    检测从机应答信号
  * @param    None
  * @retval   bool: true-从机应答, false-从机未应答
  */
bool IIC_IsSlaveAck(void)
{
    bool ack;
    
    /* 设置SDA为输入模式，准备读取从机应答 */
    SDA_Config(GPIO_Mode_IN);
    
    /* 拉低SCL，从机准备应答位 */
    SCL_SET(0);
    delay_us(5);

    /* 拉高SCL，主机读取应答状态 */
    SCL_SET(1);
    delay_us(5);

    /* 读取SDA状态：1表示未应答，0表示已应答 */
    if(SDA_READ)
        ack = false;  // 未应答(NACK)
    else
        ack = true;   // 已应答(ACK)

    /* 拉低SCL，结束应答周期 */
    SCL_SET(0);
    delay_us(5);
    
    return ack;
}

/**
  * @function IIC_Stop 
  * @brief    IIC停止信号
  * @param    None
  * @retval   None
  */
void IIC_Stop(void)
{
    /* 设置SDA为输出模式 */
    SDA_Config(GPIO_Mode_OUT);
    
    /* 确保SDA和SCL为低电平 */
    SCL_SET(0);
    SDA_SET(0);
    delay_us(5);

    /* 先拉高SCL */
    SCL_SET(1);
    delay_us(5);
    
    /* 在SCL为高时，SDA从低变高产生停止信号 */
    SDA_SET(1);
    delay_us(5);
}

/**
  * @function Write_IIC_Command 
  * @brief    写命令到OLED
  * @param    IIC_Command: 要写入的命令
  * @retval   None
  */
void Write_IIC_Command(unsigned char IIC_Command)
{
    IIC_Start();
    IIC_SendByte(0x78);           // 从机地址，SA0=0
    IIC_IsSlaveAck();
    IIC_SendByte(0x00);           // 写命令标识
    IIC_IsSlaveAck();
    IIC_SendByte(IIC_Command);    // 发送命令
    IIC_IsSlaveAck();
    IIC_Stop();
}

/**
  * @function Write_IIC_Data 
  * @brief    写数据到OLED
  * @param    IIC_Data: 要写入的数据
  * @retval   None
  */
void Write_IIC_Data(unsigned char IIC_Data)
{
    IIC_Start();
    IIC_SendByte(0x78);           // 从机地址，D/C#=0; R/W#=0
    IIC_IsSlaveAck();
    IIC_SendByte(0x40);           // 写数据标识
    IIC_IsSlaveAck();
    IIC_SendByte(IIC_Data);       // 发送数据
    IIC_IsSlaveAck();
    IIC_Stop();
}

/**
  * @function OLED_WR_Byte 
  * @brief    OLED写字节函数
  * @param    dat: 要写入的数据/命令
  * @param    cmd: 命令/数据标识(0-命令, 1-数据)
  * @retval   None
  */
void OLED_WR_Byte(unsigned dat, unsigned cmd)
{
    if(cmd)
    {
        Write_IIC_Data(dat);      // 写数据
    }
    else
    {
        Write_IIC_Command(dat);   // 写命令
    }
}

{
    unsigned char m, n;

        }
    }
}

/**
  * @function OLED_Set_Pos 
  * @brief    设置OLED显示位置
  * @param    x: 列坐标(0-127)
  * @param    y: 页坐标(0-7)
  * @retval   None
  */
void OLED_Set_Pos(unsigned char x, unsigned char y)
{
    OLED_WR_Byte(0xb0 + y, OLED_CMD);                 // 设置页地址
    OLED_WR_Byte(((x & 0xf0) >> 4) | 0x10, OLED_CMD); // 设置列地址高位
    OLED_WR_Byte((x & 0x0f), OLED_CMD);               // 设置列地址低位
}

  * @function OLED_Clear

void OLED_Clear(void)

    u8 i, n;
    for(i = 0; i < 8; i++)
    {

/**
  * @function OLED_ShowChar 
  * @brief    在指定位置显示一个字符
  * @param    x: 列坐标(0-127)
  * @param    y: 页坐标(0-7)
  * @param    chr: 要显示的字符
  * @param    Char_Size: 字体大小(12或16)
  * @retval   None
  */
void OLED_ShowChar(u8 x, u8 y, u8 chr, u8 Char_Size)
{
    unsigned char c = 0, i = 0;
    c = chr - ' ';  // 得到偏移后的值
    
    if(x > Max_Column - 1)
    {
        x = 0;
        y = y + 2;
    }
    
    if(Char_Size == 16)
    {
        OLED_Set_Pos(x, y);
        for(i = 0; i < 8; i++)
            OLED_WR_Byte(F8X16[c * 16 + i], OLED_DATA);
        OLED_Set_Pos(x, y + 1);
        for(i = 0; i < 8; i++)
            OLED_WR_Byte(F8X16[c * 16 + i + 8], OLED_DATA);
    }
    else
    {
        OLED_Set_Pos(x, y);
        for(i = 0; i < 6; i++)
            OLED_WR_Byte(F6x8[c][i], OLED_DATA);
    }
}

/**
  * @function oled_pow 
  * @brief    幂运算函数
  * @param    m: 底数
  * @param    n: 指数
  * @retval   u32: 计算结果
  */
u32 oled_pow(u8 m, u8 n)
{
    u32 result = 1;
    while(n--)
        result *= m;
    return result;
}

/**
  * @function OLED_ShowNum 
  * @brief    显示数字
  * @param    x: 起始列坐标
  * @param    y: 起始页坐标
  * @param    num: 要显示的数字
  * @param    len: 数字位数
  * @param    size2: 字体大小
  * @retval   None
  */
void OLED_ShowNum(u8 x, u8 y, u32 num, u8 len, u8 size2)
{
    u8 t, temp;
    u8 enshow = 0;
    
    for(t = 0; t < len; t++)
    {
        temp = (num / oled_pow(10, len - t - 1)) % 10;
        if(enshow == 0 && t < (len - 1))
        {
            if(temp == 0)
            {
                OLED_ShowChar(x + (size2 / 2) * t, y, ' ', size2);
                continue;
            }
            else
                enshow = 1;
        }
        OLED_ShowChar(x + (size2 / 2) * t, y, temp + '0', size2);
    }
}

/**
  * @function OLED_ShowString 
  * @brief    显示字符串
  * @param    x: 起始列坐标
  * @param    y: 起始页坐标
  * @param    chr: 要显示的字符串
  * @param    Char_Size: 字体大小
  * @retval   None
  */
void OLED_ShowString(u8 x, u8 y, u8 *chr, u8 Char_Size)
{
    unsigned char j = 0;
    while(chr[j] != '\0')
    {
        OLED_ShowChar(x, y, chr[j], Char_Size);
        x += 8;
        if(x > 120)
        {
            x = 0;
            y += 2;
        }
        j++;
    }
}

/**
  * @function OLED_DrawBMP 
  * @brief    显示BMP图片
  * @param    x0: 起始列坐标
  * @param    y0: 起始页坐标
  * @param    x1: 结束列坐标
  * @param    y1: 结束页坐标
  * @param    BMP: 图片数据数组
  * @retval   None
  */
void OLED_DrawBMP(unsigned char x0, unsigned char y0, unsigned char x1, unsigned char y1, unsigned char BMP[])
{
    unsigned int j = 0;
    unsigned char x, y;
    
    if(y1 % 8 == 0)
        y = y1 / 8;
    else
        y = y1 / 8 + 1;
        
    for(y = y0; y < y1; y++)
    {
        OLED_Set_Pos(x0, y);
        for(x = x0; x < x1; x++)
        {
            OLED_WR_Byte(BMP[j++], OLED_DATA);
        }
    }
}

/**

            SCL--PG7
            SDA--PG6
            KEY1--PA0
            KEY2--PE2

#include "stm32f4xx.h"
#include "delay.h"
#include "oled.h"
#include <stdio.h>
#include <stdbool.h>
#include <bmp.h>

/**
  * @function Key1_Config 
  * @brief    按键1初始化(PA0)
  * @param    None
  * @retval   None
  */
void Key1_Config(void)
{
    GPIO_InitTypeDef GPIO_InitStructure;
    
    /* 使能GPIOA端口时钟 */
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOA, ENABLE);
    
    /* 配置PA0引脚 */
    GPIO_InitStructure.GPIO_Pin = GPIO_Pin_0;        // 引脚编号
    GPIO_InitStructure.GPIO_Mode = GPIO_Mode_IN;     // 输入模式
    GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_NOPULL; // 无上下拉
    GPIO_Init(GPIOA, &GPIO_InitStructure);
}

/**
  * @function Key2_Config 
  * @brief    按键2初始化(PE2)
  * @param    None
  * @retval   None
  */
void Key2_Config(void)
{
    GPIO_InitTypeDef GPIO_InitStructure;
    
    /* 使能GPIOE端口时钟 */
    RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOE, ENABLE);
    
    /* 配置PE2引脚 */
    GPIO_InitStructure.GPIO_Pin = GPIO_Pin_2;        // 引脚编号
    GPIO_InitStructure.GPIO_Mode = GPIO_Mode_IN;     // 输入模式
    GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_NOPULL; // 无上下拉
    GPIO_Init(GPIOE, &GPIO_InitStructure);
}

/**
  * @function main 
  * @brief    主函数
  * @param    None
  * @retval   int: 程序执行状态
  */
int main(void)
{
    /* 硬件初始化 */
    Key1_Config();      // 初始化按键1
    Key2_Config();      // 初始化按键2
    OLED_Init();        // 初始化OLED
    OLED_Clear();       // 清屏
    
    /* 主循环 */
    while(1)
    {
        /* 检测按键1按下 */
        if(GPIO_ReadInputDataBit(GPIOA, GPIO_Pin_0))
        {
            /* 显示BMP1图片 */
            OLED_DrawBMP(0, 0, 128, 8, BMP1);
            delay_ms(500);  // 防抖延时
        }
        
        /* 检测按键2按下 */
        if(GPIO_ReadInputDataBit(GPIOE, GPIO_Pin_2))
        {
            /* 显示BMP2图片 */
            OLED_DrawBMP(0, 0, 128, 8, BMP2);
            delay_ms(500);  // 防抖延时
        }
    }
}
