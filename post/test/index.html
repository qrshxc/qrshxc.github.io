void SeqList_Insert(SeqList *L, int x) {
    int temp = -1; // 记录待插入元素的下标
    
    // 遍历顺序表，找到插入位置，比较元素

        if (x < L[i]) {
            temp = i;
            break;
        }
    }

    if (-1 == temp) {
        L[last+1] = x;
        return;
    }
    
    // 把待插入位置的后继元素向后移动

    L[temp] = x; // 插入当前位置
}
